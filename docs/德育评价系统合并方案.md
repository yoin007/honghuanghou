### 2.3 权限检查函数

```python
def check_permission(user, permission: str) -> bool:
    """检查用户是否有某权限"""
    if not user:
        return False

    role = user.role if hasattr(user, 'role') else user.get('role', '')
    role_config = ROLES.get(role, {})

    # admin拥有所有权限
    if 'all' in role_config.get('permissions', []):
        return True

    return permission in role_config.get('permissions', [])


def check_class_access(user, class_id: int) -> bool:
    """检查用户是否有班级访问权限"""
    if check_permission(user, 'report_view_all'):
        return True

    if check_permission(user, 'report_view_own_class'):
        # 班主任只能访问自己的班级
        teacher_class = get_teacher_class(user.teacher_id)
        return teacher_class == class_id

    return False
```

## 三、数据合并方案

### 3.1 合并原则

```
┌─────────────────────────────────────────────────────────────┐
│                    合并原则                                  │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  1. 最小改动原则                                            │
│     ├── 保留现有系统核心功能                                │
│     ├── 新增德育模块，不破坏现有逻辑                        │
│     └── 兼容现有数据格式                                    │
│                                                             │
│  2. 数据统一原则                                            │
│     ├── 班级、学生、教师数据统一管理                        │
│     ├── 一处维护，多处使用                                  │
│     └── 保持数据一致性                                      │
│                                                             │
│  3. 渐进式迁移原则                                          │
│     ├── Excel→数据库 分步迁移                               │
│     ├── 保留过渡期兼容                                      │
│     └── 无缝切换                                            │
│                                                             │
│  4. 权限统一原则                                            │
│     ├── 统一角色定义                                        │
│     ├── 统一认证机制                                        │
│     └── 细粒度权限控制                                      │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 3.2 方案选择

#### 方案A：统一到数据库（推荐）

```
优势：
- 数据结构清晰，便于维护
- 支持复杂查询和统计
- 事务支持，数据一致性好
- 便于德育评价功能实现

劣势：
- 需要迁移现有Excel数据
- 需要调整部分现有代码
```

#### 方案B：保持Excel + 新增德育数据库

```
优势：
- 现有系统改动最小
- 快速上线

劣势：
- 数据分散，维护困难
- 数据一致性难保证
- 德育统计查询效率低

不推荐原因：
- 德育评价涉及大量统计计算
- Excel不适合频繁写入和复杂查询
```

**结论：推荐方案A，统一到数据库**

### 5.2 迁移脚本

```python
# scripts/migrate_data.py
#
# 注意：以下代码使用 MySQL 占位符 %s
# 如使用 SQLite，请将 %s 替换为 ?

import pandas as pd
from datetime import date

class DataMigrator:
    """数据迁移工具"""

    def __init__(self, excel_path: str, db):
        self.excel_path = excel_path
        self.db = db

    def migrate_all(self):
        """执行完整迁移"""
        print("=" * 60)
        print("开始数据迁移...")
        print("=" * 60)

        # 1. 迁移级号
        self.migrate_grades()

        # 2. 迁移班级
        self.migrate_classes()

        # 3. 迁移学生
        self.migrate_students()

        # 4. 迁移教师
        self.migrate_teachers()

        print("\n" + "=" * 60)
        print("数据迁移完成！")
        print("=" * 60)

    def migrate_grades(self):
        """从班级代码提取级号"""
        print("\n[1] 迁移级号数据...")

        df = pd.read_excel(self.excel_path, sheet_name='class')

        # 从class_code提取级号（如202501 → 2025级）
        class_codes = df['class_code'].astype(str).tolist()
        years = set()
        for code in class_codes:
            if len(code) >= 4:
                years.add(int(code[:4]))

        count = 0
        for year in sorted(years):
            self.db.execute("""
                INSERT INTO grade (grade_name, enrollment_year)
                VALUES (%s, %s)
                ON DUPLICATE KEY UPDATE grade_name = VALUES(grade_name)
            """, f"{year}级", year)
            count += 1

        print(f"  迁移了 {count} 个级号")

    def migrate_classes(self):
        """迁移班级数据"""
        print("\n[2] 迁移班级数据...")

        df = pd.read_excel(self.excel_path, sheet_name='class')

        count = 0
        for _, row in df.iterrows():
            class_code = str(row['class_code'])
            class_name = str(row['class_name'])

            # 解析级号和班号
            if len(class_code) >= 6:
                year = int(class_code[:4])
                class_number = int(class_code[4:])
            else:
                continue

            # 获取级号ID
            grade = self.db.query(
                "SELECT grade_id FROM grade WHERE enrollment_year = %s", year
            ).first()

            if not grade:
                continue

            # 插入班级
            self.db.execute("""
                INSERT INTO class (class_code, grade_id, class_number, class_name,
                                   leader_name, is_active)
                VALUES (%s, %s, %s, %s, %s, %s)
                ON DUPLICATE KEY UPDATE class_name = VALUES(class_name)
            """, class_code, grade['grade_id'], class_number, class_name,
                 row.get('leaders'), row.get('active', 1))

            count += 1

        print(f"  迁移了 {count} 个班级")

    def migrate_students(self):
        """迁移学生数据"""
        print("\n[3] 迁移学生数据...")

        df = pd.read_excel(self.excel_path, sheet_name='students')

        count = 0
        errors = []

        for _, row in df.iterrows():
            sid = str(row['sid'])
            name = str(row['name'])
            cname = str(row.get('cname', ''))
            roomid = str(row.get('roomid', ''))

            # 根据班级名称查找班级
            cls = self.db.query(
                "SELECT class_id, grade_id FROM class WHERE class_name = %s", cname
            ).first()

            if not cls:
                # 尝试从sid解析班级
                if len(sid) >= 6:
                    class_code = sid[:6]
                    cls = self.db.query(
                        "SELECT class_id, grade_id FROM class WHERE class_code = %s", class_code
                    ).first()

            if not cls:
                errors.append(f"学生 {sid}({name}) 找不到班级")
                continue

            # 插入学生
            self.db.execute("""
                INSERT INTO student (student_id, name, class_id, grade_id, roomid,
                                     original_grade_id, status, is_active)
                VALUES (%s, %s, %s, %s, %s, %s, '在校', 1)
                ON DUPLICATE KEY UPDATE name = VALUES(name), class_id = VALUES(class_id)
            """, sid, name, cls['class_id'], cls['grade_id'], roomid, cls['grade_id'])

            # 创建班级履历
            self.db.execute("""
                INSERT INTO student_class_history
                (student_id, class_id, grade_id, start_date, change_reason)
                VALUES (%s, %s, %s, %s, '入学')
                ON DUPLICATE KEY UPDATE end_date = NULL
            """, sid, cls['class_id'], cls['grade_id'], '2025-09-01')

            count += 1

        print(f"  迁移了 {count} 个学生")
        if errors:
            print(f"  ⚠️ {len(errors)} 个学生迁移失败")
            for e in errors[:5]:
                print(f"     - {e}")

    def migrate_teachers(self):
        """迁移教师数据"""
        print("\n[4] 迁移教师数据...")

        df = pd.read_excel(self.excel_path, sheet_name='teachers')

        count = 0
        for _, row in df.iterrows():
            name = str(row['name'])
            subject = str(row.get('subject', ''))
            role = str(row.get('role', 'teacher'))
            level = int(row.get('level', 0))
            active = int(row.get('active', 1))
            pwd_changed = int(row.get('is_password_changed', 0))
            pwd = str(row.get('pwd', ''))

            # 生成教师ID
            teacher_id = f"T{name}"

            # 密码哈希
            if pwd and pwd_changed:
                password_hash = hash_password(pwd)
            else:
                password_hash = hash_password('123456')

            self.db.execute("""
                INSERT INTO teacher (teacher_id, name, subject, password_hash,
                                     role, level, is_active, is_password_changed)
                VALUES (%s, %s, %s, %s, %s, %s, %s, %s)
                ON DUPLICATE KEY UPDATE name = VALUES(name), role = VALUES(role)
            """, teacher_id, name, subject, password_hash, role, level, active, pwd_changed)

            count += 1

        print(f"  迁移了 {count} 个教师")


def hash_password(password: str) -> str:
    """密码哈希"""
    from passlib.context import CryptContext
    pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")
    return pwd_context.hash(password)


if __name__ == "__main__":
    # 执行迁移
    migrator = DataMigrator('checkTemplate.xlsx', db)
    migrator.migrate_all()
```

```python
def validate_migration(db):
    """验证迁移结果"""

    print("=" * 60)
    print("数据迁移验证")
    print("=" * 60)

    errors = []

    # 1. 数据量验证
    print("\n[1] 数据量验证")

    checks = [
        ("级号", "grade", "SELECT COUNT(*) as cnt FROM grade"),
        ("班级", "class", "SELECT COUNT(*) as cnt FROM class"),
        ("学生", "student", "SELECT COUNT(*) as cnt FROM student"),
        ("教师", "teacher", "SELECT COUNT(*) as cnt FROM teacher"),
    ]

    for name, table, query in checks:
        result = db.query(query).first()
        count = result['cnt']
        status = "✅" if count > 0 else "❌"
        print(f"  {status} {name}: {count} 条")
        if count == 0:
            errors.append(f"{name}数据为空")

    # 2. 关联验证
    print("\n[2] 关联验证")

    # 学生-班级关联
    orphan = db.query("""
        SELECT COUNT(*) as cnt FROM student s
        LEFT JOIN class c ON s.class_id = c.class_id
        WHERE c.class_id IS NULL
    """).first()['cnt']
    print(f"  {'✅' if orphan == 0 else '❌'} 学生-班级关联: {orphan} 条异常")

    # 班级履历验证
    no_history = db.query("""
        SELECT COUNT(*) as cnt FROM student s
        LEFT JOIN student_class_history sch ON s.student_id = sch.student_id
        WHERE s.status = '在校' AND sch.id IS NULL
    """).first()['cnt']
    print(f"  {'✅' if no_history == 0 else '❌'} 班级履历: {no_history} 人缺失")

    # 3. 结果
    print("\n" + "=" * 60)
    if not errors:
        print("✅ 验证通过")
    else:
        print("❌ 发现问题:")
        for e in errors:
            print(f"   - {e}")

    return errors
```

### 6.2.3 权限检查工具函数

```javascript
// utils/permission.js

import store from '@/store'

/**
 * 检查用户是否有指定权限
 * @param {Object} user 用户对象（可选，默认从store获取）
 * @param {string} permission 权限标识
 * @returns {boolean}
 */
export function checkPermission(user, permission) {
  // 如果只传一个参数，则从store获取用户
  if (typeof user === 'string') {
    permission = user
    user = store.state.user
  }

  // 管理员拥有所有权限
  if (user.role === 'admin') {
    return true
  }

  // admin 权限标识只对管理员开放
  if (permission === 'admin') {
    return false
  }

  // 获取角色权限配置（与后端ROLES定义保持一致）
  const rolePermissions = {
    jiaowu: [
      'teacher_manage', 'teaching_quality_monitor', 'academic_diagnosis',
      'ai_consultation', 'report_view_all', 'class_manage', 'schedule_manage',
      'birthday_reminder', 'student_profile', 'moral_record_manage', 'event_type_manage'
    ],
    xuefa: [
      'moral_record_manage', 'punishment_manage', 'event_type_manage',
      'class_change_approve', 'report_view_all', 'student_manage', 'ai_consultation',
      'birthday_reminder', 'student_profile'
    ],
    cleader: [
      'moral_record_own_class', 'report_view_own_class', 'homework_publish',
      'announcement_publish', 'leave_approve', 'ai_consultation_own_class',
      'student_profile_own_class', 'birthday_reminder_own_class'
    ],
    teacher: [
      'homework_publish', 'schedule_view', 'student_view', 'moral_record_input'
    ],
    student: [
      'moral_self_view', 'homework_view', 'schedule_view', 'profile_self_view',
      'birthday_blessing_receive'
    ],
    parent: [
      'moral_child_view', 'profile_child_view', 'birthday_reminder_child'
    ]
  }

  const permissions = rolePermissions[user.role] || []
  return permissions.includes(permission)
}

/**
 * 检查用户是否有任一权限
 * @param {string[]} permissions 权限标识数组
 * @returns {boolean}
 */
export function hasAnyPermission(permissions) {
  return permissions.some(p => checkPermission(p))
}
```

## 十、总结

### 合并要点

1. **数据统一**：Excel → 数据库，一处维护多处使用
2. **权限统一**：扩展角色体系，细粒度权限控制
3. **API兼容**：保留现有API，新增德育API
4. **渐进迁移**：分步迁移，充分验证

### 预期效果

- 现有功能保持不变
- 新增德育评价功能
- 数据管理更规范
- 权限控制更精细
- 系统可扩展性增强
